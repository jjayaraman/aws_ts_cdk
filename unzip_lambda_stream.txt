import { S3Client, GetObjectCommand } from '@aws-sdk/client-s3'
import { Upload } from '@aws-sdk/lib-storage'
import { Readable } from 'stream'
import * as unzipper from 'unzipper'

const s3Client = new S3Client({})

export const handler = async (event: any) => {
  const bucketName = event.Records[0].s3.bucket.name
  const key = decodeURIComponent(
    event.Records[0].s3.object.key.replace(/\+/g, ' ')
  )

  console.log(`Processing object: ${key} from bucket: ${bucketName}`)

  try {
    const { Body } = await s3Client.send(
      new GetObjectCommand({ Bucket: bucketName, Key: key })
    )

    if (!Body) {
      throw new Error('S3 object Body is null or undefined.')
    }

    // Ensure Body is a Readable stream
    const bodyStream =
      Body instanceof Readable ? Body : Readable.from(Body as any)

    console.log('Created readable stream from S3 Body.')

    const unzipStream = bodyStream.pipe(unzipper.Parse())

    unzipStream.on('error', (err) => {
      console.error('Unzip stream error:', err)
    })

    for await (const entry of unzipStream) {
      console.log(`Processing entry: ${entry.path} (type: ${entry.type})`)
      try {
        if (entry.type === 'File') {
          // Use Upload for optimized stream handling
          const upload = new Upload({
            client: s3Client,
            params: {
              Bucket: bucketName,
              Key: `${key.replace(/\.zip$/, '')}/${entry.path}`,
              Body: entry,
            },
          })

          await upload.done()
          console.log(`Uploaded file: ${entry.path}`)
        } else {
          console.log(`Skipping non-file entry: ${entry.path}`)
          entry.autodrain() // Drain non-file entries
        }
      } catch (entryError) {
        console.error(`Error processing entry ${entry.path}:`, entryError)
        entry.autodrain() // Ensure the stream is drained even on errors
      }
    }

    console.log(`Finished unzipping ${key}`)
    return { statusCode: 200, body: 'Unzip successful' }
  } catch (error) {
    console.error('Error:', error)
    return { statusCode: 500, body: `Error: ${error}` }
  }
}
